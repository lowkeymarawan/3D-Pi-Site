function WebsiteIntro({ onComplete }) {
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return undefined;

    let frameId;
    let disposed = false;
    const clock  = new THREE.Clock();
    const startTime = performance.now();

    // ── Renderer ─────────────────────────────────────────────────────────────
    const renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: false, powerPreference: 'high-performance' });
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    renderer.setSize(window.innerWidth, window.innerHeight);
    renderer.outputColorSpace = THREE.SRGBColorSpace;
    renderer.toneMapping = THREE.ACESFilmicToneMapping;
    renderer.toneMappingExposure = 1.18;
    renderer.shadowMap.enabled = true;
    renderer.shadowMap.type = THREE.PCFSoftShadowMap;

    // ── Post-processing (manual, no EffectComposer) ──────────────────────────
    // One HDR render target → subtle bloom + soft vignette + fade shader.
    // Intentionally restrained to keep the white/black palette clean.

    const RT = new THREE.WebGLRenderTarget(window.innerWidth, window.innerHeight, {
      minFilter: THREE.LinearFilter,
      magFilter: THREE.LinearFilter,
      format: THREE.RGBAFormat,
      type: THREE.HalfFloatType,
    });
    const RT_BLOOM = new THREE.WebGLRenderTarget(
      Math.floor(window.innerWidth / 3),
      Math.floor(window.innerHeight / 3),
      { minFilter: THREE.LinearFilter, magFilter: THREE.LinearFilter, format: THREE.RGBAFormat }
    );

    const quadGeo = new THREE.PlaneGeometry(2, 2);

    // Downsample + threshold: only very bright whites glow
    const bloomExtractMat = new THREE.ShaderMaterial({
      uniforms: { tDiffuse: { value: RT.texture }, uThreshold: { value: 0.72 } },
      vertexShader:   `varying vec2 vUv; void main(){ vUv=uv; gl_Position=vec4(position,1.); }`,
      fragmentShader: `
        uniform sampler2D tDiffuse;
        uniform float uThreshold;
        varying vec2 vUv;
        void main(){
          vec4 c = texture2D(tDiffuse, vUv);
          float lum = dot(c.rgb, vec3(0.2126,0.7152,0.0722));
          float soft = smoothstep(uThreshold, uThreshold+0.18, lum);
          gl_FragColor = vec4(c.rgb * soft, 1.);
        }`,
    });

    // Separable Gaussian blur — horizontal pass
    const blurHMat = new THREE.ShaderMaterial({
      uniforms: { tDiffuse: { value: RT_BLOOM.texture }, uDir: { value: new THREE.Vector2(1, 0) }, uRes: { value: new THREE.Vector2() } },
      vertexShader:   `varying vec2 vUv; void main(){ vUv=uv; gl_Position=vec4(position,1.); }`,
      fragmentShader: `
        uniform sampler2D tDiffuse;
        uniform vec2 uDir;
        uniform vec2 uRes;
        varying vec2 vUv;
        void main(){
          vec2 step = uDir / uRes;
          vec4 c = vec4(0.);
          c += texture2D(tDiffuse, vUv - step*3.) * 0.0625;
          c += texture2D(tDiffuse, vUv - step*2.) * 0.125;
          c += texture2D(tDiffuse, vUv - step*1.) * 0.25;
          c += texture2D(tDiffuse, vUv          ) * 0.25;
          c += texture2D(tDiffuse, vUv + step*1.) * 0.25;
          c += texture2D(tDiffuse, vUv + step*2.) * 0.125;
          c += texture2D(tDiffuse, vUv + step*3.) * 0.0625;
          gl_FragColor = c;
        }`,
    });

    // Final composite
    const compositeMat = new THREE.ShaderMaterial({
      uniforms: {
        tScene: { value: RT.texture },
        tBloom: { value: RT_BLOOM.texture },
        uFade:  { value: 0 },
        uTime:  { value: 0 },
      },
      vertexShader: `varying vec2 vUv; void main(){ vUv=uv; gl_Position=vec4(position,1.); }`,
      fragmentShader: `
        uniform sampler2D tScene;
        uniform sampler2D tBloom;
        uniform float uFade;
        uniform float uTime;
        varying vec2 vUv;

        float hash(vec2 p){ return fract(sin(dot(p,vec2(127.1,311.7)))*43758.5); }

        void main(){
          vec4 scene = texture2D(tScene, vUv);
          vec4 bloom = texture2D(tBloom, vUv);

          // Additive bloom — only affects bright areas, so background stays clean
          scene.rgb += bloom.rgb * 0.28;

          // Very soft vignette — tasteful, not crushing
          vec2  c   = vUv - 0.5;
          float vig = 1.0 - smoothstep(0.38, 1.1, length(c) * 1.55);
          scene.rgb *= mix(0.78, 1.0, vig);

          // Film grain — extremely subtle
          float grain = (hash(vUv + fract(uTime * 0.017)) - 0.5) * 0.015;
          scene.rgb += grain;

          // Master fade (black → scene → black)
          scene.rgb *= uFade;

          gl_FragColor = vec4(scene.rgb, 1.);
        }`,

    const rim = new THREE.DirectionalLight(0xffffff, 2.2);
    rim.position.set(-2, 6, -7);

    const nozzleGlow = new THREE.PointLight(0xffd4a0, 0, 3, 2);

    bed.receiveShadow = true;

    scene.add(railX);

    railXFront.position.z = 2.2;

    const crossRod = new THREE.Mesh(new THREE.CylinderGeometry(0.042, 0.042, 4.5, 28), railMat);
    crossRod.position.set(0, 3.25, 0);
    crossRod.castShadow = true;
    scene.add(crossRod);

    // 4 corner posts
    [-3.6, 3.6].forEach((x) => {
      [-2.2, 2.2].forEach((z) => {

    });

    // ── Nozzle Carriage ──────────────────────────────────────────────────────
    const nozzleGroup = new THREE.Group();
    scene.add(nozzleGroup);

    body.castShadow = true;
    nozzleGroup.add(body);

    const screwGeo = new THREE.CylinderGeometry(0.05, 0.05, 0.022, 24);
    [[-0.22, -0.42], [-0.22, -0.68], [0.22, -0.42], [0.22, -0.68]].forEach(([x, y]) => {

    });

    // Heatsink fins (6 fins, evenly spaced)
    const finMat = new THREE.MeshStandardMaterial({ color: 0x555555, roughness: 0.22, metalness: 0.78 });
    for (let i = 0; i < 6; i++) {

    // Heater block — matte dark

    const heater = new THREE.Mesh(new THREE.BoxGeometry(0.44, 0.26, 0.38), heaterMat);
    heater.position.y = -1.24;
    heater.castShadow = true;
    nozzleGroup.add(heater);

    // Nozzle tip — polished steel
    const tipMat = new THREE.MeshStandardMaterial({ color: 0x999999, roughness: 0.08, metalness: 0.96 });
    const tip = new THREE.Mesh(new THREE.ConeGeometry(0.175, 0.36, 32), tipMat);
    tip.rotation.x = Math.PI;
    tip.position.y = -1.5;
    tip.castShadow = true;
    nozzleGroup.add(tip);

    // Filament strand — dark, thin
    const filament = new THREE.Mesh(
      new THREE.CylinderGeometry(0.02, 0.02, 1.0, 10),
      new THREE.MeshStandardMaterial({ color: 0x080808, roughness: 0.55, transparent: true, opacity: 0 })
    );
    filament.position.y = -2.0;
    nozzleGroup.add(filament);

    // ── Logo Geometry ─────────────────────────────────────────────────────────
    const logoGroup = new THREE.Group();
    logoGroup.position.set(0, 0.105, 0);
    scene.add(logoGroup);

    const LOGO_SCALE  = 0.038;
    const STROKE_RAD  = 0.115;

    function svgPoint(sx, sy, y = 0) {
      return new THREE.Vector3((sx - 100) * LOGO_SCALE, y, (sy - 100) * LOGO_SCALE);
    }

    function arcPoints(cx, cy, r, startDeg, endDeg, steps = 130) {
      const pts = [];
      for (let i = 0; i <= steps; i++) {
        const deg = startDeg + (endDeg - startDeg) * (i / steps);
        const rad = (deg * Math.PI) / 180;
        pts.push(svgPoint(cx + r * Math.cos(rad), cy + r * Math.sin(rad)));
      }
      return pts;
    }

    function tubeFromCurve(pts, radius = STROKE_RAD, segs = 130) {
      const curve = new THREE.CatmullRomCurve3(pts, false, 'catmullrom', 0.05);
      return new THREE.TubeGeometry(curve, segs, radius, 20, false);
    }

    function tubeFromPolyline(svgPts, radius = STROKE_RAD, segs = 80) {
      const path = new THREE.CurvePath();
      const pts  = svgPts.map(([sx, sy]) => svgPoint(sx, sy));
      for (let i = 0; i < pts.length - 1; i++) path.add(new THREE.LineCurve3(pts[i], pts[i + 1]));
      return new THREE.TubeGeometry(path, segs, radius, 20, false);
    }

    // Polished matte-black for the printed logo — like a fresh FDM print
    const logoTopMat = new THREE.MeshStandardMaterial({
      color:     0x0d0d0d,
      roughness: 0.52,
      metalness: 0.10,
      transparent: true,
      opacity:   0,
    });
    const logoSideMat = new THREE.MeshStandardMaterial({
      color:     0x050505,
      roughness: 0.70,
      metalness: 0.05,
      transparent: true,
      opacity:   0,
    });

    const circleGeo  = tubeFromCurve(arcPoints(100, 100, 82, 292, 608, 160), STROKE_RAD, 190);
    const circleMesh = new THREE.Mesh(circleGeo, logoTopMat.clone());
    circleMesh.castShadow = circleMesh.receiveShadow = true;
    logoGroup.add(circleMesh);

      tubeFromPolyline([[69, 24], [100, 104], [131, 24]], STROKE_RAD, 100),

      c.castShadow = c.receiveShadow = true;

    scene.add(trailPts);
    let trailHead = 0, trailCount = 0, trailAccum = 0;
    const trailLastPos = new THREE.Vector3();

    // ── Nozzle path ───────────────────────────────────────────────────────────
    const BED_TOP_Y   =  0.105;

    const PRINT_Y     =  BED_TOP_Y + 0.055 - TIP_LOCAL_Y;
    const TRAVEL_Y    =  PRINT_Y + 0.44;

    function nozzlePoint(sx, sy, y = PRINT_Y) {
      const p = svgPoint(sx, sy);
      return new THREE.Vector3(p.x, y, p.z);
    }

    const waypoints = [];
    waypoints.push(nozzlePoint(126, 22, TRAVEL_Y));

    arcPoints(100, 100, 82, 292, 608, 100).forEach(p => waypoints.push(new THREE.Vector3(p.x, PRINT_Y, p.z)));

    waypoints.push(nozzlePoint(69, 24, PRINT_Y));

    waypoints.push(nozzlePoint(74, 16, PRINT_Y));
    [[74,16],[100,10],[126,16],[100,52],[74,16]].forEach(([sx,sy]) => waypoints.push(nozzlePoint(sx,sy,PRINT_Y)));
    waypoints.push(nozzlePoint(150, -10, TRAVEL_Y));

    const nozzleCurve = new THREE.CatmullRomCurve3(waypoints, false, 'catmullrom', 0.12);

    // ── Camera rig — smoothly damped ──────────────────────────────────────────
    // We never snap the camera. Every camera property is lerped or smoothstepped.

    const camPosA  = new THREE.Vector3(4.8, 2.35, 7.2);   // low-angle tracking
    const camTarA  = new THREE.Vector3(0.0, 0.7,  0.0);
    const camPosB  = new THREE.Vector3(0.0, 10.8, 0.35);  // bird's-eye reveal
    const camTarB  = new THREE.Vector3(0.0, 0.05, 0.0);

    // Smoothly interpolated camera state
    const camPosCurrent = camPosA.clone();
    const camTarCurrent = camTarA.clone();

    const T = {
      printEnd:    4.35,
      liftStart:   3.8,
      liftEnd:     6.2,
      done:        13.2,
    };

    // Logo stagger — same window as original
    const logoReveal = [
      { mesh: circleMesh, start: 0.45, end: 2.55 },
      { mesh: circleDepth,start: 0.55, end: 2.65 },
      { mesh: vMesh,      start: 2.05, end: 3.15 },
      { mesh: vDepth,     start: 2.15, end: 3.25 },
      { mesh: sliceMesh,  start: 3.05, end: 4.0  },
      { mesh: sliceDepth, start: 3.15, end: 4.1  },
    ];

    // ── Easing helpers ────────────────────────────────────────────────────────
    const clamp01     = v => Math.max(0, Math.min(1, v));
    const smoothstep  = t => t * t * (3 - 2 * t);
    const smootherstep = t => t * t * t * (t * (t * 6 - 15) + 10);  // C2-continuous

    // ── Resize ────────────────────────────────────────────────────────────────
    function onResize() {
      const w = window.innerWidth, h = window.innerHeight;
      camera.aspect = w / h;
      camera.updateProjectionMatrix();
      renderer.setSize(w, h);
      RT.setSize(w, h);
      RT_BLOOM.setSize(Math.floor(w / 3), Math.floor(h / 3));
      blurHMat.uniforms.uRes.value.set(Math.floor(w / 3), Math.floor(h / 3));
    }

    // ── Animate ───────────────────────────────────────────────────────────────
    function animate() {
      if (disposed) return;
      frameId = requestAnimationFrame(animate);

      const delta   = Math.min(clock.getDelta(), 0.05);  // cap delta to avoid huge jumps on tab-switch
      const elapsed = (performance.now() - startTime) / 1000;

      // ── Nozzle movement ──
      // smootherstep gives a buttery eased-in / eased-out print motion
      const printT  = clamp01(elapsed / T.printEnd);
      const pathT   = smootherstep(printT);
      nozzleGroup.position.copy(nozzleCurve.getPointAt(Math.min(pathT, 0.9999)));

      // Look-ahead for orientation — larger step = smoother but lags corners.
      // 0.008 is the sweet spot for this curve density.
      const aheadT = Math.min(pathT + 0.008, 0.9999);
      const lookPt = nozzleCurve.getPointAt(aheadT);
      nozzleGroup.lookAt(lookPt.x, nozzleGroup.position.y, lookPt.z);

      // ── Logo fade-in ──
      logoReveal.forEach(({ mesh, start, end }) => {
        mesh.material.opacity = smoothstep(clamp01((elapsed - start) / (end - start)));
      });

      // ── Filament + nozzle glow ──
      const tipDist    = Math.abs(nozzleGroup.position.y - PRINT_Y);
      const isPrinting = tipDist < 0.55 && elapsed < T.printEnd && elapsed > 0.5;

      if (isPrinting) {
        filament.material.opacity = 0.88;
        filament.scale.y = Math.max(0.08, 1 - tipDist * 1.8);
        // Gentle pulse — low frequency so it doesn't look choppy
        nozzleGlow.intensity = 2.8 + Math.sin(elapsed * 11) * 0.55;
        nozzleGlow.position.copy(nozzleGroup.position);
        nozzleGlow.position.y -= 0.15;
      } else {
        filament.material.opacity = Math.max(0, filament.material.opacity - delta * 4);
        // Smooth glow fade-out — exponential decay, not instant cut
        nozzleGlow.intensity = Math.max(0, nozzleGlow.intensity - delta * 9);
      }

      // ── Sparks ──
      // Only emit during active printing, fewer sparks, white only
      sparks.forEach((sp, idx) => {
        sp.life += delta;
        if (sp.life > sp.maxLife) {
          sp.life = 0;
          sp.maxLife = 0.25 + Math.random() * 0.18;
          const angle = Math.random() * Math.PI * 2;
          const speed = 0.035 + Math.random() * 0.04;
          sp.velocity.set(Math.cos(angle) * speed, 0.025 + Math.random() * 0.05, Math.sin(angle) * speed);
          sp.mesh.position.set((Math.random() - 0.5) * 0.07, -1.58, (Math.random() - 0.5) * 0.07);
        }
        const active = isPrinting && (idx % 3 === Math.floor(elapsed * 9) % 3);
        if (active) {
          sp.mesh.position.addScaledVector(sp.velocity, delta * 55);
          sp.velocity.y -= delta * 0.22;  // gentle gravity
          sp.mesh.material.opacity = (1 - sp.life / sp.maxLife) * 0.82;
        } else {
          sp.mesh.material.opacity = Math.max(0, sp.mesh.material.opacity - delta * 8);
        }
      });

      // ── Extrusion trail ──
      trailAccum += delta;
      if (isPrinting && trailAccum > 0.014) {
        trailAccum = 0;
        const wp = new THREE.Vector3(nozzleGroup.position.x, BED_TOP_Y + 0.062, nozzleGroup.position.z);
        if (wp.distanceTo(trailLastPos) > 0.018) {
          trailLastPos.copy(wp);
          const i = trailHead * 3;
          trailPos[i] = wp.x; trailPos[i+1] = wp.y; trailPos[i+2] = wp.z;
          // Monochrome: hot = near white, cooled = dark grey
          const age = trailCount / TRAIL_MAX;
          const brightness = 0.85 - age * 0.6;
          trailCol[i] = brightness; trailCol[i+1] = brightness; trailCol[i+2] = brightness;
          trailHead  = (trailHead + 1) % TRAIL_MAX;
          trailCount = Math.min(trailCount + 1, TRAIL_MAX);
          trailGeo.setDrawRange(0, trailCount);
          trailGeo.attributes.position.needsUpdate = true;
          trailGeo.attributes.color.needsUpdate    = true;
        }
      }

      // Clean up temporary extrusion dots after the printed logo is complete.
      // Prevents any stray center shaft/artifact during the final hold.
      if (elapsed > T.printEnd + 0.25) {
        trailMat.opacity = Math.max(0, trailMat.opacity - delta * 1.8);
      }

      // ── Camera — fully damped, never snaps ──
      let targetPos, targetLook;

      if (elapsed < T.liftStart) {
        // Act 1: low-angle follow cam. Gently biases toward the nozzle tip.
        const nozzleFocus = new THREE.Vector3(
          nozzleGroup.position.x,
          Math.max(0.45, nozzleGroup.position.y - 1.85),
          nozzleGroup.position.z
        );
        targetPos  = camPosA;
        // Mix between fixed look-target and nozzle focus — 40% nozzle bias
        targetLook = new THREE.Vector3().lerpVectors(camTarA, nozzleFocus, 0.4);
      } else {
        // Act 2+: crane lift to bird's-eye.
        // smootherstep (C2) ensures the lift starts and ends without any pop.
        const lt = smootherstep(clamp01((elapsed - T.liftStart) / (T.liftEnd - T.liftStart)));
        targetPos  = new THREE.Vector3().lerpVectors(camPosA, camPosB, lt);
        targetLook = new THREE.Vector3().lerpVectors(camTarA, camTarB, lt);
      }

      // Exponential smooth-damp — framerate-independent, always converges
      // alpha = 1 - e^(-stiffness * delta)
      const stiffness = 5.0;
      const alpha     = 1 - Math.exp(-stiffness * delta);
      camPosCurrent.lerp(targetPos,  alpha);
      camTarCurrent.lerp(targetLook, alpha);
      camera.position.copy(camPosCurrent);
      camera.lookAt(camTarCurrent);

      // ── Post-processing fade ──
      // Fade in over first 0.9 s, fade out over last 0.5 s
      const fadeIn  = smoothstep(clamp01(elapsed / 0.9));
      const fadeOut = smoothstep(clamp01(1 - (elapsed - (T.done - 0.5)) / 0.5));
      compositeMat.uniforms.uFade.value = fadeIn * (elapsed > T.done - 0.5 ? fadeOut : 1);
      compositeMat.uniforms.uTime.value = elapsed;

      // ── Render pipeline ──
      // Pass 1: full scene → HDR target
      renderer.setRenderTarget(RT);
      renderer.render(scene, camera);

      // Pass 2: threshold extract → bloom buffer
      postScene.add(bloomExtractQuad);
      bloomExtractMat.uniforms.tDiffuse.value = RT.texture;
      renderer.setRenderTarget(RT_BLOOM);
      renderer.render(postScene, postCam);
      postScene.remove(bloomExtractQuad);

      // Pass 3: blur bloom (horizontal)
      postScene.add(blurQuad);
      blurHMat.uniforms.tDiffuse.value = RT_BLOOM.texture;
      blurHMat.uniforms.uDir.value.set(1, 0);
      renderer.setRenderTarget(RT_BLOOM);
      renderer.render(postScene, postCam);
      postScene.remove(blurQuad);

      // Pass 4: composite to screen
      postScene.add(compositeQuad);
      compositeMat.uniforms.tScene.value = RT.texture;
      compositeMat.uniforms.tBloom.value = RT_BLOOM.texture;
      renderer.setRenderTarget(null);
      renderer.render(postScene, postCam);
      postScene.remove(compositeQuad);

      if (elapsed > T.done) {
        onComplete?.();
        disposed = true;
      }
    }

    window.addEventListener('resize', onResize);
    onResize();
    animate();

    return () => {
      disposed = true;
      window.removeEventListener('resize', onResize);
      cancelAnimationFrame(frameId);
      logoMeshes.forEach(m => { m.geometry?.dispose(); m.material?.dispose(); });
      trailGeo.dispose();
      RT.dispose(); RT_BLOOM.dispose();
      renderer.dispose();
    };
  }, [onComplete]);

  return (
    <div className="three-intro" aria-hidden="true">
      <canvas id="intro-canvas" ref={canvasRef} />
      <div className="intro-logo-morph">
        <BrandLogo />
      </div>
    </div>
  );
}
